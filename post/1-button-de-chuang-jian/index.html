1. Button的创建

#include <QPushButton>  // 引入头文件

// 第一种方式

// 创建一个按钮
QPushButton * btn = new QPushButton;
btn->show();    // 以顶层方式弹出窗口控件，也就是游离于主窗口之外
// 让btn对象依赖在MainWindow窗口中
btn->setParent(this);

// 显示文本
btn->setText("set");


// 第二种方式

// 创建第二个按钮，按照控件的大小创建窗口
QPushButton * btn2 = new QPushButton("the first", this);
// 移动btn2按钮
btn2->move(50, 0);

// 重置主窗口大小
resize(400, 300);

// 设置固定的主窗口大小
setFixedSize(400, 300);


//设置主窗口标题
setWindowTitle("第一个窗口");
