2.连续存储数组的算法演示

2020-01-09
#include <stdio.h>
#include <stdlib.h>
#include <malloc.h>

struct Arr
{
	int *pBase;	// 存储的是数组第一个元素的地址
	int len;	// 数组所能容纳的最大元素的地址
	int cnt;	// 当前数组有效元素的个数
};

void init_arr(struct Arr * pArr, int length);	
bool append_arr(struct Arr * pArr, int val);
bool insert_arr(struct Arr * pArr, int pos, int val);	// pos的值从1开始
bool delete_arr(struct Arr * pArr, int pos);
int get();
bool is_empty(struct Arr * pArr);
bool is_full(struct Arr * pArr);
void sort_arr(struct Arr * pArr);
void show_arr(struct Arr * pArr);
void inversion_arr();


int main()
{
	struct Arr arr;
	init_arr(&arr, 9);
	append_arr(&arr, 1);
	append_arr(&arr, 2);
	append_arr(&arr, 9);
	append_arr(&arr, 4);
	append_arr(&arr, 5);
	append_arr(&arr, 6);
	//append_arr(&arr, 7);
	//append_arr(&arr, 8);
	//insert_arr(&arr, 0, 88);

	//delete_arr(&arr, 2);
	sort_arr(&arr);
	show_arr(&arr);




	printf("\n\n\n\n");
	return 0;
}


// @brief 数组初始化
// @param 结构体的地址
// @param 数组的长度
void init_arr(struct Arr * pArr, int length)
{
	pArr->pBase = (int *)malloc(sizeof(int) * length);
	if(pArr->pBase == NULL)
	{
		printf("动态内存分配失败\n");
		exit(-1);	// 终止整个程序
	}
	
	printf("动态内存分配成功!\n");
	pArr->len = length;
	pArr->cnt = 0;

	return ;
}


// @brief 判度是否已满
//
//
bool is_full(struct Arr * pArr)
{
	return pArr->cnt == pArr->len ? true : false;
}

// @brief 追加元素
//
//
bool append_arr(struct Arr * pArr, int val)
{
	// 数组为满的判断
	if(is_full(pArr))
	{
		printf("数组已满!\n");
		exit(-1);
	}

	// 不满时追加 
	pArr->pBase[pArr->cnt] = val;
	(pArr->cnt)++;
	return true;
}


// @brief 插入元素
//
//
bool insert_arr(struct Arr * pArr, int pos, int val)
{
	// 判断下标合法性
	if(pos < 0)
	{
		printf("插入元素的下标非法!\n");
		exit(-1);
	}

	// 先对元素进行移动,后面的元素先移动
	(pArr->cnt)++;
	for(int i = pArr->cnt - 1; i >= pos; i--)
	{
		pArr->pBase[i] = pArr->pBase[i - 1];
	}
	// 元素赋值
	pArr->pBase[pos] = val;

	return true;
}


// @brief 删除特定位置的数值
//
//
bool delete_arr(struct Arr * pArr, int pos)
{
	if(pos < 0)
	{
		printf("删除的元素下标有误!\n");
		exit(-1);
	}

	for(int i = pos; i < pArr->cnt; i++)
	{
		pArr->pBase[i] = pArr->pBase[i + 1];
	}
	(pArr->cnt)--;

	return true;
}




// @brief 判断数组是否为空
//
//
bool is_empty(struct Arr * pArr)
{
	return pArr->cnt == 0 ? true : false;	// 使用三元运算符
}


// @brief 对数组排序
//
//
void sort_arr(struct Arr * pArr)
{
	int temp;
	for(int i = 0; i < pArr->cnt - 1; i++)
	{
		for(int j = 0; j < pArr->cnt - i - 1; j++)
		{
			if(pArr->pBase[j] > pArr->pBase[j + 1])
			{
				temp = pArr->pBase[j];
				pArr->pBase[j] = pArr->pBase[j + 1];
				pArr->pBase[j + 1] = temp;
			}
		}
	}
}




// @brief 输出数组的元素
//
//
void show_arr(struct Arr * pArr)
{
	if(is_empty(pArr) == true)
	{
		printf("数组为空!\n");
		exit(-1);
	}

	for(int i = 0; i < pArr->cnt; i++)
	{
		printf("%d ", pArr->pBase[i]);
	}
}
