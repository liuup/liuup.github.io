#include <QMenu>

QMenuBar * bar = menuBar();
setMenuBar(bar);
// 创建菜单
QMenu * fileMenu = bar->addMenu("File");
QMenu * editMenu = bar->addMenu("Edit");
QMenu * helpMenu = bar->addMenu("Help");

// 创建菜单项
fileMenu->addAction("New file");
fileMenu->addAction("NULL");
fileMenu->addSeparator();
QAction *file_exit = fileMenu->addAction("Exit");
connect(file_exit, &QAction::triggered, this, &MainWindow::close);

QAction *help_about = helpMenu->addAction("About");
connect(help_about, &QAction::triggered, this, &MainWindow::HelpAboutFunc);
