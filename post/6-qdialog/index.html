6. QDialog

2020-01-09

对话框

// 点击新建按钮,弹出一个对话框
connect(ui->actionNew, &QAction::triggered, this, [=](){
    // 对话框 分离
    // 模态对话框(不可以对其他窗口进行操作)
    // 非模态对话框(可以对其他窗口进行操作)

    // 模态
    QDialog newWin(this);
    newWin.resize(200, 100);
    newWin.exec();
    cout << "模态弹出" << endl;

    // 非模态对话框
    QDialog * newWin2 = new QDialog(this);
    newWin2->resize(200, 150);
    newWin2->show();
    // 为了防止内存泄漏,需要移除对象
    newWin2->setAttribute(Qt::WA_DeleteOnClose);    // 55号属性

    
    cout << "非模态对话框" << endl;



});
